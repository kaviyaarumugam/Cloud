Calculator.java

public class Calculator
{
    public int add(int a,int b)
    {
        return a + b;
    }
    
    public int sub(int a,int b)
    {
        return a - b;
    }
    
    public int mul(int a,int b){
        return a*b;
    }
    
    public int div(int a,int b){
        return a/b;
    }
    
    public int mod(int a,int b){
        return a%b;
    }
    
    public int sq(int a){
        return (int)Math.sqrt(a);
    }
}
